#for语句

#基本语法（参考C/C++)

for(语句1;语句2;语句3){
    执行代码;
}

#嵌套循环

规则

  1. 外层为假时内层不循环；
  2. 先执行外层再执行内层，当内层条件为假时再返回外层执行。

举个栗子

//打印九九乘法表
for(var i=1;i<=9;i++){
    for(var j=1;j<=i;j++){
        document.write(j+"*"+i+"="+j*i+"&nbsp;");
        document.write("<br/>")
    }
}

/*
 功能：弹出输入框
 返回值：
 1. 点击确定，返回输入内容；

/* 功能：判断一个值是否不是数字
  返回值：布尔值
  注意：NaN和所有值都不相等，包括它自己

length
/*  
    语法：string.length

    返回值：number */

new Date().getDay()
/*  功能：获取星期
    返回值：number(0-6)
    */
